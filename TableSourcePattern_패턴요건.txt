## 소스 테이블 기준
1. FROM 절에 테이블이 1개
INSERT INTO DM.`요약01`
SELECT TIME_START
, ENT_ID AS entId
, APPL_NAME AS `송신	`
FROM DW.`기지국01` AS N1
WHERE N1.COL1 = 'A'

2.LEFT JOIN 또는 LEFT OUTER JOIN에서 소스테이블 추출
SELECT TIME_START
, ENT_ID AS entId
, APPL_NAME AS `송신	`
FROM DW.`기지국02` AS N1
LEFT JOIN DW.`부서02` AS N2
ON N1.ENT_ID = N2.ENT_ID
WHERE N1.COL1 = 'A'

3.JOIN 또는 INNER JOIN에서 소스테이블 추출
SELECT TIME_START
, ENT_ID AS entId
, APPL_NAME AS `송신	`
FROM DW.`기지국03` AS N1
LEFT JOIN DW.`부서03` AS N2
ON N1.ENT_ID = N2.ENT_ID
WHERE N1.COL1 = 'A'

4.Oracle JOIN 문법에서 , 다음의 테이블 추출
SELECT TIME_START
, ENT_ID AS entId
, APPL_NAME AS `송신	`
FROM DW.`기지국04` AS N1
 , DW.`부서04` AS N2
WHERE N1.ENT_ID = N2.ENT_ID
AND N1.COL1 = 'A'

5. WITH 절안에 테이블만 추출
INSERT INTO DM.`요약01`
WITH `모수`  AS
(
SELECT TIME_START
, ENT_ID AS entId
, APPL_NAME AS `송신	`
FROM DW.`기지국05` AS N1
WHERE N1.COL1 = 'A'
) ,
MOSU2 AS
(
SELECT TIME_START
, ENT_ID AS entId
, APPL_NAME AS `송신	`
FROM DW.`부서05` AS N1
WHERE N1.COL1 = 'A'
)
SELECT TIME_START
, ENT_ID AS entId
, APPL_NAME AS `송신	`
FROM `모수`
